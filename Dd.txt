```c
#include <stdio.h>
#include <string.h>
#include <ctype.h>

char keywords[][20] = {
    "int","float","char","if","else","while","for","return",
    "double","void","break","continue","switch","case","default",
    "do","long","short","unsigned","signed","const","static","struct"
};

int keywordCount = sizeof(keywords) / sizeof(keywords[0]);

int isKeyword(char word[])
{
    for(int i = 0; i < keywordCount; i++)
    {
        if(strcmp(word, keywords[i]) == 0)
            return 1;
    }
    return 0;
}

int main()
{
    char input[5000], line[200];
    char token[100];
    int i = 0, j = 0;

    input[0] = '\0';

    printf("Enter C program (Type END on a new line to finish):\n");

    while(1)
    {
        fgets(line, sizeof(line), stdin);

        if(strncmp(line, "END", 3) == 0)
            break;

        strcat(input, line);
    }

    printf("\n----- Lexical Analysis -----\n");

    while(input[i] != '\0')
    {
        if(isspace(input[i]))
        {
            i++;
            continue;
        }

        if(isalpha(input[i]) || input[i] == '_')
        {
            j = 0;
            while(isalnum(input[i]) || input[i] == '_')
            {
                token[j++] = input[i++];
            }
            token[j] = '\0';

            if(isKeyword(token))
                printf("%-15s : Keyword\n", token);
            else
                printf("%-15s : Identifier\n", token);
        }
        else if(isdigit(input[i]))
        {
            j = 0;
            while(isdigit(input[i]) || input[i] == '.')
            {
                token[j++] = input[i++];
            }
            token[j] = '\0';

            printf("%-15s : Number\n", token);
        }
        else
        {
            switch(input[i])
            {
                case '+':
                case '-':
                case '*':
                case '/':
                case '%':
                case '=':
                case '<':
                case '>':
                case '!':
                    printf("%c               : Operator\n", input[i]);
                    break;

                case ';':
                case ',':
                case '(':
                case ')':
                case '{':
                case '}':
                case '[':
                case ']':
                    printf("%c               : Special Symbol\n", input[i]);
                    break;

                default:
                    printf("%c               : Unknown\n", input[i]);
            }
            i++;
        }
    }

    return 0;
}
```
